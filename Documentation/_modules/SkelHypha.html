"""    
:Notes:
    
    This script has been tested with the following dependencies 
    (other versions may work):
        
        * Python 3.6.8, 3.7.2 and 3.7.3
        * numpy 1.16.2 and 1.16.3
        * skimage (scikit-image) 0.14.2 and 0.15.0
        * tifffile 0.15.1 and 2019.3.18
    
    If the program takes several minutes to compute the skeleton of the slice 0 
    of the first image, check that the matrix of the binarized image is of type 
    uint8 with the background at 0 value (not 255!). If it's at 255, try 
    switching on/off the line 114 which inverts the image.

.. moduleauthor:: Laura Xénard
"""

import os
import errno
import time
import math

import numpy as np
import tifffile as tiff
import skimage.util as skiu
import skimage.morphology as skim


def startTimer():
    return time.time()
    
def endTimer(start):
    end = time.time()
    return end - start


[docs]def isTif(path):
    """ 
    Checks that the element of path 'path' is a tif/tiff image.
    
    :param str path: the path of a file/directory
    :return: True if it's a tif/tiff, False otherwise
    :rtype: bool
    """
    
    if os.path.isfile(path):
        formats = ['.tif', '.tiff']
        fileFormat = os.path.splitext(path)[1] # extension of the file
        if fileFormat in formats:
            return True
        else:
            return False
    else:
        return False



[docs]def createOutputFolder(folder_in):
    """
    Create the output directory that will hold the skeleton images and the log file.
    
    :param str folder_in: the path of the directory containing the images to skeletonized
    :return: the path that will hold the skeleton images and the path of the log file
    :rtype: (str, str)        
    """
     
    folder_out = os.path.abspath(folder_in + '/Skeletons')
    log_path = os.path.join(folder_out, 'log.txt')
    try :
        os.mkdir(folder_out)
    except OSError as exception:
        if exception.errno != errno.EEXIST: # no exception if the directory already exists
            print('An exception has occured while creating the directory: errno {}'.format(exception.errno))
            raise
        pass
    
    return (folder_out, log_path)



[docs]def loadImg(path):
    """
    Load an image and convert it to the correct format.
    
    :param str path: the path of the image to load and convert
    :return: the converted image
    :rtype: np.ndarray
    """
    
    img = tiff.imread(path)
    if img.dtype == bool:                   
        img = skiu.img_as_ubyte(img) # converting to uint8 if boolean image
    img = skiu.invert(img) # to use if the background is at 255 instead of 0
    
    return img



[docs]def sliceImg(img):
    """
    Divide the image in 25 slices of nearly identical size.
    
    :param img: the image to be sliced
    :type img: np.ndarray
    :return: the list of slices and the processing time of the slicing
    :rtype: (list[np.ndarray], int)   
    """              

    print('   Slicing the image in 25...')
    start_sli = startTimer()
    slice_height = math.ceil(img.shape[0]/5)
    slice_width = math.ceil(img.shape[1]/5)                    
    
    # Next line comes from: https://stackoverflow.com/questions/5953373/how-to-split-image-into-multiple-pieces-in-python, Nir answer
    slices = [img[x:x+slice_height, y:y+slice_width] for x in range(0, img.shape[0], slice_height) for y in range(0, img.shape[1], slice_width)] 

    processing_time_sli = endTimer(start_sli)                   
    print('   ...slicing complete, done in {:.3f} s.'.format(processing_time_sli))
                    
    return (slices, processing_time_sli)



[docs]def createSubImages(slices):
    """
    Create 25 subimages made of 3x3 slices. The skeleton will be computed on 3x3 slices, 
    but only the core slice will be kept for the final skeleton (to avoid skeleton errors when at the border of an image).
    Completely black slices (void) are added when necessary to create a 3x3 slices for each previously created slice.
    
    :param slices: the slices that will be used to create the subimages
    :type slices: list[np.ndarray]
    :return: a list of the subimages created and the processing time of the computing
    :rtype: (list[np.ndarray], int)
    """
    
    print('   Computing subimages...')
    start_comp = startTimer()
    subimages = [] # list holding all the subimages to skeletonize 
    void_slice = np.zeros_like(slices[0]) # black slice to add when necessary
    
    # Line 1
    
    subimg0 = np.concatenate((np.concatenate((void_slice, void_slice, void_slice), axis = 1), 
                              np.concatenate((void_slice, slices[0], slices[1]), axis = 1), 
                              np.concatenate((void_slice, slices[5], slices[6]), axis = 1)), axis = 0)
    
    subimg1 = np.concatenate((np.concatenate((void_slice, void_slice, void_slice), axis = 1), 
                              np.concatenate((slices[0], slices[1], slices[2]), axis = 1), 
                              np.concatenate((slices[5], slices[6], slices[7]), axis = 1)), axis = 0)
    
    subimg2 = np.concatenate((np.concatenate((void_slice, void_slice, void_slice), axis = 1), 
                              np.concatenate((slices[1], slices[2], slices[3]), axis = 1), 
                              np.concatenate((slices[6], slices[7], slices[8]), axis = 1)), axis = 0)
    
    subimg3 = np.concatenate((np.concatenate((void_slice, void_slice, np.zeros_like(slices[4])), axis = 1), 
                              np.concatenate((slices[2], slices[3], slices[4]), axis = 1), 
                              np.concatenate((slices[7], slices[8], slices[9]), axis = 1)), axis = 0)
    
    subimg4 = np.concatenate((np.concatenate((void_slice, np.zeros_like(slices[4]), void_slice), axis = 1), 
                              np.concatenate((slices[3], slices[4], void_slice), axis = 1), 
                              np.concatenate((slices[8], slices[9], void_slice), axis = 1)), axis = 0)
    
    subimages.extend([subimg0, subimg1, subimg2, subimg3, subimg4])
    
    # Line 2
    
    subimg5 = np.concatenate((np.concatenate((void_slice, slices[0], slices[1]), axis = 1), 
                              np.concatenate((void_slice, slices[5], slices[6]), axis = 1), 
                              np.concatenate((void_slice, slices[10], slices[11]), axis = 1)), axis = 0)
    
    subimg6 = np.concatenate((np.concatenate((slices[0], slices[1], slices[2]), axis = 1), 
                              np.concatenate((slices[5], slices[6], slices[7]), axis = 1), 
                              np.concatenate((slices[10], slices[11], slices[12]), axis = 1)), axis = 0)
    
    subimg7 = np.concatenate((np.concatenate((slices[1], slices[2], slices[3]), axis = 1), 
                              np.concatenate((slices[6], slices[7], slices[8]), axis = 1), 
                              np.concatenate((slices[11], slices[12], slices[13]), axis = 1)), axis = 0)
    
    subimg8 = np.concatenate((np.concatenate((slices[2], slices[3], slices[4]), axis = 1), 
                              np.concatenate((slices[7], slices[8], slices[9]), axis = 1), 
                              np.concatenate((slices[12], slices[13], slices[14]), axis = 1)), axis = 0)
    
    subimg9 = np.concatenate((np.concatenate((slices[3], slices[4], void_slice), axis = 1), 
                              np.concatenate((slices[8], slices[9], void_slice), axis = 1), 
                              np.concatenate((slices[13], slices[14], void_slice), axis = 1)), axis = 0)
    
    subimages.extend([subimg5, subimg6, subimg7, subimg8, subimg9])
    
    # Line 3
    
    subimg10 = np.concatenate((np.concatenate((void_slice, slices[5], slices[6]), axis = 1), 
                               np.concatenate((void_slice, slices[10], slices[11]), axis = 1), 
                               np.concatenate((void_slice, slices[15], slices[16]), axis = 1)), axis = 0)
    
    subimg11 = np.concatenate((np.concatenate((slices[5], slices[6], slices[7]), axis = 1), 
                               np.concatenate((slices[10], slices[11], slices[12]), axis = 1), 
                               np.concatenate((slices[15], slices[16], slices[17]), axis = 1)), axis = 0)
    
    subimg12 = np.concatenate((np.concatenate((slices[6], slices[7], slices[8]), axis = 1), 
                               np.concatenate((slices[11], slices[12], slices[13]), axis = 1), 
                               np.concatenate((slices[16], slices[17], slices[18]), axis = 1)), axis = 0)
    
    subimg13 = np.concatenate((np.concatenate((slices[7], slices[8], slices[9]), axis = 1), 
                               np.concatenate((slices[12], slices[13], slices[14]), axis = 1), 
                               np.concatenate((slices[17], slices[18], slices[19]), axis = 1)), axis = 0)
    
    subimg14 = np.concatenate((np.concatenate((slices[8], slices[9], void_slice), axis = 1), 
                               np.concatenate((slices[13], slices[14], void_slice), axis = 1), 
                               np.concatenate((slices[18], slices[19], void_slice), axis = 1)), axis = 0)
    
    subimages.extend([subimg10, subimg11, subimg12, subimg13, subimg14])
    
    # Line 4
    
    subimg15 = np.concatenate((np.concatenate((void_slice, slices[10], slices[11]), axis = 1), 
                               np.concatenate((void_slice, slices[15], slices[16]), axis = 1), 
                               np.concatenate((np.zeros_like(slices[20]), slices[20], slices[21]), axis = 1)), axis = 0)
    
    subimg16 = np.concatenate((np.concatenate((slices[10], slices[11], slices[12]), axis = 1), 
                               np.concatenate((slices[15], slices[16], slices[17]), axis = 1), 
                               np.concatenate((slices[20], slices[21], slices[22]), axis = 1)), axis = 0)
    
    subimg17 = np.concatenate((np.concatenate((slices[11], slices[12], slices[13]), axis = 1), 
                               np.concatenate((slices[16], slices[17], slices[18]), axis = 1), 
                               np.concatenate((slices[21], slices[22], slices[23]), axis = 1)), axis = 0)       

    subimg18 = np.concatenate((np.concatenate((slices[12], slices[13], slices[14]), axis = 1), 
                               np.concatenate((slices[17], slices[18], slices[19]), axis = 1), 
                               np.concatenate((slices[22], slices[23], slices[24]), axis = 1)), axis = 0)
    
    subimg19 = np.concatenate((np.concatenate((slices[13], slices[14], void_slice), axis = 1), 
                               np.concatenate((slices[18], slices[19], void_slice), axis = 1), 
                               np.concatenate((slices[23], slices[24], np.zeros_like(slices[23])), axis = 1)), axis = 0)
    
    subimages.extend([subimg15, subimg16, subimg17, subimg18, subimg19])
    
    # Line 5
    
    subimg20 = np.concatenate((np.concatenate((void_slice, slices[15], slices[16]), axis = 1), 
                               np.concatenate((np.zeros_like(slices[20]), slices[20], slices[21]), axis = 1), 
                               np.concatenate((void_slice, void_slice, void_slice), axis = 1)), axis = 0)
    
    subimg21 = np.concatenate((np.concatenate((slices[15], slices[16], slices[17]), axis = 1), 
                               np.concatenate((slices[20], slices[21], slices[22]), axis = 1), 
                               np.concatenate((void_slice, void_slice, void_slice), axis = 1)), axis = 0)
    
    subimg22 = np.concatenate((np.concatenate((slices[16], slices[17], slices[18]), axis = 1), 
                               np.concatenate((slices[21], slices[22], slices[23]), axis = 1), 
                               np.concatenate((void_slice, void_slice, void_slice), axis = 1)), axis = 0)       

    subimg23 = np.concatenate((np.concatenate((slices[17], slices[18], slices[19]), axis = 1), 
                               np.concatenate((slices[22], slices[23], slices[24]), axis = 1), 
                               np.concatenate((void_slice, void_slice, np.zeros_like(slices[19])), axis = 1)), axis = 0)
    
    subimg24 = np.concatenate((np.concatenate((slices[18], slices[19], void_slice), axis = 1), 
                               np.concatenate((slices[23], slices[24], np.zeros_like(slices[23])), axis = 1), 
                               np.concatenate((void_slice, np.zeros_like(slices[19]), void_slice), axis = 1)), axis = 0)
    
    subimages.extend([subimg20, subimg21, subimg22, subimg23, subimg24]) 
    
    processing_time_comp = endTimer(start_comp)
    print('   ...computing complete, done in {:.3f} s.'.format(processing_time_comp))
    
    return (subimages, processing_time_comp)



[docs]def skeletonizeSubimages(subimages):
    """
    Apply a skeletonization (morphological thinning) on every subimage of the list.
    
    :param subimages: the list of the subimages to skeletonize
    :type subimages: list[np.ndarray]
    :return: the list of the skeletonized subimages and the processing time of the skeletonization
    :rtype: (list[np.ndarray], int)      
    """
    
    print('   Beginning skeletonization...')
    start_skel = startTimer()
    skels = [] # list holding all the partial skeletons
    
    for index, image in enumerate(subimages):
        skel = skim.thin(image) # returns a boolean image
        skel = skiu.img_as_ubyte(skel, force_copy=False) # reconverting to uint8 image
        skels.append(skel)        
        print('     ...slice {}/25 done...'.format(index+1))
    
    processing_time_skel = endTimer(start_skel)
    print('   ...skeletonization complete, done in {:.0f} min {:.3f} s.'.format(processing_time_skel // 60, processing_time_skel % 60))
    
    return (skels, processing_time_skel)



[docs]def reconstructSkel(skels, img):
    """
    Reconstruct the skeleton image by concatenating the core slice of every subimage.
    
    :param skels: the list of the skeletonized subimages
    :type skels: list[np.ndarray]
    :param img: the original image
    :type img: np.ndarray
    :return: the final image and the processing time of the reconstruction
    :rtype: (np.ndarray, int)
    """
        
    print('   Beginning skeleton reconstruction...')
    start_rec = startTimer()
    
    slice_height = math.ceil(img.shape[0]/5)
    slice_width = math.ceil(img.shape[1]/5)
    
    final_skel_line1 = np.concatenate((skels[0][slice_height : slice_height*2, slice_width : slice_width*2], 
                                       skels[1][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[2][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[3][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[4][slice_height : slice_height*2, slice_width : slice_width*2]), axis = 1)
    
    final_skel_line2 = np.concatenate((skels[5][slice_height : slice_height*2, slice_width : slice_width*2], 
                                       skels[6][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[7][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[8][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[9][slice_height : slice_height*2, slice_width : slice_width*2]), axis = 1)
    
    final_skel_line3 = np.concatenate((skels[10][slice_height : slice_height*2, slice_width : slice_width*2], 
                                       skels[11][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[12][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[13][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[14][slice_height : slice_height*2, slice_width : slice_width*2]), axis = 1)
    
    final_skel_line4 = np.concatenate((skels[15][slice_height : slice_height*2, slice_width : slice_width*2], 
                                       skels[16][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[17][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[18][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[19][slice_height : slice_height*2, slice_width : slice_width*2]), axis = 1)
    
    final_skel_line5 = np.concatenate((skels[20][slice_height : slice_height*2, slice_width : slice_width*2], 
                                       skels[21][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[22][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[23][slice_height : slice_height*2, slice_width : slice_width*2],
                                       skels[24][slice_height : slice_height*2, slice_width : slice_width*2]), axis = 1)
    
    final_skel = np.concatenate((final_skel_line1, final_skel_line2, final_skel_line3, final_skel_line4, final_skel_line5), axis =0)
        
    # Cropping the image to compensate for the added pixels when the size of the binarized image is not a multiple of 5
    final_skel = final_skel[:img.shape[0], :img.shape[1]]
    
    processing_time_rec = endTimer(start_rec)
    print('   ...reconstruction complete, done in {:.3f} s.'.format(processing_time_rec))
    
    return (final_skel, processing_time_rec)

def main():

    path = input('Path to the directory containing the images to skeletonize, with / or \: \n'
                 '(example: C:/Users/Documents/Binarized_img)\n')   
    start = time.time() # to get the processing time    
    folder_in = os.path.abspath(path) # cross platform, / or \ doesn't matter
    img_count = 0 # total number of images to process
    current_img = 0 # img currently being processed
    
    try:    
        with os.scandir(folder_in) as dirIt:
            for entry in(dirIt):
                if isTif(entry.path):
                    img_count += 1
    except FileNotFoundError:
        print('ERROR! Folder not found.') 
    except NotADirectoryError:
        print('ERROR! The path given is not a folder.')
    
    folder_out, log_path = createOutputFolder(folder_in)
        
    # Processing the images, one by one
    try:    
        with os.scandir(folder_in) as dirIt:
            for entry in(dirIt):
                if isTif(entry.path):
                    
                    current_img += 1
                    img_name = os.path.splitext(entry.name)[0]
                    img_ext = os.path.splitext(entry.name)[1]
                    path_out = os.path.abspath(os.path.join(folder_out, img_name + '_skel' + img_ext)) # output image path
                    
                    print('Creating skeleton for image {} ({} of {})...'.format(img_name, current_img, img_count))
                    start_img = startTimer()                                                    
                    img = loadImg(entry.path) # loading the image and converting it to the correct format
                    slices, processing_time_sli  = sliceImg(img) # slicing the image in 25
                    subimages, processing_time_sub = createSubImages(slices) # creating subimages from 3x3 slices
                    skels, processing_time_skel = skeletonizeSubimages(subimages) # computing the skeleton on each subimages
                    final_skel, processing_time_rec = reconstructSkel(skels, img) # reconstructing the final image from the core slice of every image
                    tiff.imsave(path_out, final_skel) # saving the skeleton image                    
                    processing_time_img = endTimer(start_img)
                    print('...skeleton {} created, done in {:.0f} min {:.3f} s.'.format(img_name, processing_time_img // 60, processing_time_img % 60))
                    print('***')
                    
                    # Log creation/update
                    txt = '*** Processing of image {} ({} of {}) ***\n'.format(img_name, current_img, img_count)
                    txt += '    Slicing done in {:.3f} s.\n'.format(processing_time_sli)
                    txt += '    Creation of subimages done in {:.3f} s.\n'.format(processing_time_sub)
                    txt += '    Skeletonization done in {:.0f} min {:.3f} s.\n'.format(processing_time_skel // 60, processing_time_skel % 60)
                    txt += '    Image reconstruction done in {:.3f} s.\n'.format(processing_time_rec)
                    txt += '  Total processing time for {}: {:.0f} min {:.3f} s.\n'.format(img_name, processing_time_img // 60, processing_time_img % 60)                                    
                    try:
                        with open(log_path, 'a') as log:
                            log.write(txt)
                            log.write('\n')
                    except OSError:
                        print('OSError while writing the log at image {}.'.format(img_name))
    except FileNotFoundError:
        print('ERROR! Folder not found.') 
    except NotADirectoryError:
        print('ERROR! The path given is not a folder.')
    
    processing_time = endTimer(start)
    minutes = processing_time // 60
    seconds = processing_time % 60
    txt = 'Total processing time for {} image(s): {:.0f} min {:.3f} s.'.format(img_count, minutes, seconds)
    print(txt)
    
    # Log final line
    try:
        with open(log_path, 'a') as log:
            log.write(txt)
    except OSError:
        print('OSError while writing the log final line.')
        
if __name__ == '__main__':
    main()
